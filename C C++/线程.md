#### 创建一个不传参的线程



```c++
#include <pthread.h>
#include <iostream>
#include <unistd.h>

using namespace std;

void *func(void * arg)
{
    cout<<"I am child thread"<<endl;
    return (void*)0;
}


int main()
{
    
    pthread_t tidp;
    int ret;

    ret = pthread_create(&tidp,NULL,func,NULL);

    if (ret)
    {
        cout <<"pthread create failed:"<< ret <<endl;
    }
    
    sleep(1); //如果没有行代码 main线程会先执行结束，子线程没有机会执行，程序就结束了

    cout<<"in main:thread "<<endl;

    return 0;
}

结果：

I am child thread
in main:thread
```
